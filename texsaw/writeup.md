# 2. Tính bảng Hamming Weight (số bit 1) từ 0-255 để lookup cho nhanh
HW = np.array([bin(i).count('1') for i in range(256)])

# 3. Load data
traces = np.load('traces.npy')
plaintexts = np.load('plaintexts.npy')

num_traces, num_points = traces.shape
key = []

print("[+] Bắt đầu tấn công CPA...")

# Lặp qua 16 bytes của khóa
for b in range(16):
    max_corr = 0
    best_guess = 0
    
    # Lấy cột tương ứng của byte đang xét trong tất cả plaintexts
    pt_col = plaintexts[:, b]
    
    # Thử 256 giá trị có thể có của 1 byte khóa (0x00 đến 0xFF)
    for guess in range(256):
        # Tính năng lượng giả định (Hamming Weight của đầu ra S-box)
        hyp = HW[sbox[pt_col ^ guess]]
        
        # --- Tính Correlation Pearson giữa hyp và traces ---
        hyp_mean = np.mean(hyp)
        traces_mean = np.mean(traces, axis=0)
        
        # Công thức: covariance / (std_hyp * std_traces)
        numerator = np.sum((hyp[:, None] - hyp_mean) * (traces - traces_mean), axis=0)
        denominator = np.sqrt(np.sum((hyp - hyp_mean)**2) * np.sum((traces - traces_mean)**2, axis=0))
        
        # Bỏ qua lỗi chia cho 0 nếu có
        correlations = np.abs(numerator / (denominator + 1e-10))
        
        # Lấy đỉnh tương quan cao nhất
        peak = np.max(correlations)
        if peak > max_corr:
            max_corr = peak
            best_guess = guess
            
    key.append(best_guess)
    print(f"Byte {b:02d}: {hex(best_guess):<4} (Độ tương quan: {max_corr:.4f})")

print(f"\n[+] Khóa AES tìm được (hex): {bytes(key).hex()}")
